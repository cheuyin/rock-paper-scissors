// Computer randomly chooses "rock", "papers", or "scissors"
Function computerChoice
    Create variable randNum
    Create variable computerChoice
    Choose random number 1-3
    Assign random number to randNum
    If randNum equals 1
        computerChoice equals "rock"
    Else if randNum equals 2
        computerChoice equals "paper"
    Else
        computerChoice equals "scissors"
    return computerChoice

// Ask for player's choice 
Function playerChoice
    Create variable playerInput
    Ask player to enter "rock", "paper", or "scissors"
    Remove extraneous spaces from input and convert it to lowercase
    Store input into playerInput
    Check if playerInput is valid with checkPlayerInput()
    If playerInput is valid
        return playerInput
    Else   
        return false
